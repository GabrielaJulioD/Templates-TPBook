Alerts

Displays an alert dialog with optional content specified. The information is an optional value of the text to be displayed in the alert dialog, or, alternatively, an object to be converted to text and be shown. 

              
<!-- alert container -->

              <div
              class="cont-alert"
              style="font-family: 'Noto Sans', sans-serif; font-size: 17.5pt">
<!-- alert button action to show alert content -->

<!-- Change the text of the button inside the value -->

              <input
              value="Show live alert"
              type="button"
              style="
              text-align: center;
              margin-top: 20px;
              margin-bottom: 20px;
              min-width: 150px;
              background-color: #a01c96;
              border-radius: 4px;
              border: none;
              color: #fff;
              margin-right: 5px;
              cursor: pointer;
              width: 100px;
              padding: 10px;
              text-decoration: none;"
              onclick="document.getElementById('alert1').style.display='block';"
              />
<!-- alert content -->
              <div
              id="alert1"
              style="padding: 20px; background-color: #0087ff; color: white"
              >
<!-- close button alert -->
              <span
              class="closebtn"
              style="
              margin-left: 15px;
              color: rgb(139, 139, 238);
              font-weight: bold;
              float: right;
              font-size: 22px;
              line-height: 20px;
              cursor: pointer;
              transition: 0.3s;
              color: #fff;"
              onclick="this.parentElement.style.display='none';"
              >×</span>
<!-- alert text information -->

<!-- Change the alert information inside the tags strong and span -->

              <strong>Info!</strong> <span>Standard Alert</span>
              </div>
              </div>
