Cards

Cards are elements that allow displaying and highlighting information accompanied by images and hyperlinks. 

                
 <!-- cards contentent -->
         <div
         class="content"
         style="
           font-family: 'Noto Sans', sans-serif;
           width: 100%;
           display: flex;
           justify-content: center;
           align-items: center;
           background: none;
           margin-top: 0;
         "
       >
<!-- cards container -->
         <div
           class="container"
           style="
             max-width: fit-content;
             min-width: 200px;
             height: 430px;
             display: flex;
             flex-direction: row;
           "

<!-- first card -->

           <div
             id="card"
             style="
               width: 330px;
               height: 430px;
               float: left;
               border-radius: 4px;
               overflow: hidden;
               position: relative;
               top: 0;
               text-align: center;
               transition: all 0.25s;
               box-shadow: 0 2px 2px rgba(0, 0, 0, 0.2);
             "
             onmouseover="document.getElementById('card').style.top= '-15px'; document.getElementById('card').style.boxShadow = ' 0px 12px 16px rgba(120,0,150, 0.5)' "
             onmouseout="document.getElementById('card').style.top= '0'; document.getElementById('card').style.boxShadow = ' 0 2px 2px rgba(0, 0, 0, 0.2)' "
           >
