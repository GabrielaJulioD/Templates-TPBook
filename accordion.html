Accordion

It allows you to display different contents classified in sections, so that one of these sections remains open and closes automatically only when you open another one. 

                

<!--accordion container-->
                    <div class="accordion" style="
                    font-family: 'Noto Sans', sans-serif;
                    font-size: 17.5pt;
                    margin: 0 auto; background-color: #fff; width: 70%; border: 1px solid #ddd; border-radius: 4px; padding: 20px;">

<!--accordion section 1-->

                    <section class="accordion-item">

<!--accordion section 1 title-->


<!--Change the title with the h1 tag-->

                    <h1 id="button1" style="padding: 10px; background-color: #a01c96; color: #fff; border-radius: 4px; margin: 0; border-bottom-left-radius: 0px; border-bottom-right-radius: 0px; cursor: pointer; border: 1px solid #8d1883;"
                    onclick="document.getElementById('button1').style.background = '#8a177f'; document.getElementById('item1').style.display = 'block'; document.getElementById('item2').style.display = 'none';  document.getElementById('item3').style.display = 'none'" 
                    onpointerover="document.getElementById('item1').style.display = 'none'"
                    onmouseout="document.getElementById('button1').style.background = '#a01c96'"> Accordion Item #1 </h1>

<!--accordion section paragraph-->


<!--Change the paragraph with the p tag-->

                    <p id="item1" style="margin-top: 0px; padding: 10px; border: 1px solid #ddd; border-top: none; display: none;">

                    molestias, tenetur repellat fugiat quam dolorem mollitia, inventore
                    quibusdam unde vitae ipsam.
                    </p>
                    
                    </section>
