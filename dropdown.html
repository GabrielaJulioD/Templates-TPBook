Dropdown

A drop-down menu is a feature that will allow you to place links or drop-down menus, which will direct users to the desired content. 

                
<!-- dropdown container -->
                    <div
                    class="dropdown"
                    style="font-family: 'Noto Sans', sans-serif;
                    font-size: 17.5pt; position: relative; display: inline-block; width: 100%;"
                    onmouseover="document.getElementById('dropdown-content').style.display = 'block';"
                    onmouseout="document.getElementById('dropdown-content').style.display = 'none';"
                  >
<!-- dropdown button -->

<!-- Change the text of the button inside the value  -->

                    <input
                     value="Dropdown button"
                     type="button"
                      id="btn_popover"
                      style="
                                  text-align: center;
                                  margin-top: 20px;
                                  min-width: 150px;
                                  background-color: #a01c96;
                                  border-radius: 4px;
                                  border: none;
                                  color: #fff;
                                  margin-right: 5px;
                                  cursor: pointer;
                                  width: 100px;
                                  padding: 10px;
                                  text-decoration: none;
                      "
                      onmouseover=" document.getElementById('btn_popover').style.backgroundColor = '#8a177f'"
                      onmouseout=" document.getElementById('btn_popover').style.backgroundColor = '#a01c96'"
                     />
<!-- dropdown content -->
                    <div
                      id="dropdown-content"
                      style="
                      display: none;
                        position: absolute;
                        min-width: 150px;
                        max-width: fit-content;
                        border-radius: 4px;
                        z-index: 1;
                        background-color: #fff;
                        border: 1px solid #ddd;
                      "
                    >

<!-- action 1 -->


<!-- Change the Link action inside of the a tag  -->

                      <a
                        id="action1"
                        href="#"
                        style="
                          display: block;
                          text-decoration: none;
                          padding: 10px 10px;
                          color: #1c1c1c;
                        "
                        onmouseover="document.getElementById('action1').style.backgroundColor = '#ddd'"
                        onmouseout="document.getElementById('action1').style.backgroundColor = '#fff'"
                        > Link 1 </a>
