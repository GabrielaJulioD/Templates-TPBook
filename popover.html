Popover

The popover is a feature that will allow you to enter information to access it quickly when the user passes the mouse over the button. 

                
 <!-- popover container -->
                    <div class="dismissible_popover" style="font-family: 'Noto Sans', sans-serif;
                    font-size: 17.5pt; display: flex;"
                    onmouseover="document.getElementById('popover').style.display = 'block';"
                    onmouseout="document.getElementById('popover').style.display = 'none';">
<!-- popover button -->

<!-- Change the text of the button inside the value  -->

                      <input
                       value="Dismissible popover"
                      type="button"
                      id="btn_popover"
                      style="
                      text-align: center;
                      margin-top: 20px;
                      min-width: 150px;
                      background-color: #a01c96;
                      border-radius: 4px;
                      border: none;
                      color: #fff;
                      margin-right: 5px;
                      cursor: pointer;
                      width: 100px;
                      padding: 10px;
                      text-decoration: none;
            "
            onmouseover="document.getElementById('popover').style.display = 'block';document.getElementById('btn_popover').style.backgroundColor = '#8a177f'"
            onmouseout="document.getElementById('popover').style.display = 'none'; document.getElementById('btn_popover').style.backgroundColor = '#a01c96'"
           />
<!-- popover content -->
          <div
            id="popover"
            style="
              display: none;
              position: absolute;
              left: 200px;
              width: 250px;
              min-height: 110px;
              line-height: 23px;
              border-radius: 4px;
              z-index: 1;
              border: 1px solid #ddd;
              background-color: #fff;
            "
          >
 <!-- popover title -->

<!--Change the title with the h3 tag-->

            <h3
              class="popovertitle"
              style="
                margin: 0;
                padding: 0.5rem 1rem;
                font-size: 1rem;
                background-color: #f0f0f0;
              "
            >
              Popover title
            </h3>
<!-- popover paragraph -->

 <!-- Change the paragraph inside the p tag -->

            <p style="padding: 0.5rem 1rem; font-size: 1rem">
              Lorem ipsum dolor sit amet consectetur adipisicing elit. Tempora ab
              vitae dolore, facere nobis quia odio. Exercitationem natus debitis
              eaque voluptates est blanditiis vel, ipsa dolorum reiciendis pariatur.
              Ipsum, natus.
            </p>
          </div>
        </div>
