Toast

These are notifications whose function is to indicate a message or important information in the form of a reminder. They work through a button that shows the Toast when clicked. 

                
<!-- Toast Conatiner -->
                    <div class="toast" style="font-family: 'Noto Sans', sans-serif;
                    font-size: 17.5pt;">
<!-- Show Toast Button -->

<!-- Change the text of the button inside the value  -->

                      <input 
                      value="Show live toast"
                      type="button" class="bttn-toast" id="ToastIDBtn" style="
                      text-align: center;
                      margin-top: 20px;
                      min-width: 150px;
                      background-color: #a01c96;
                      border-radius: 4px;
                      border: none;
                      color: #fff;
                      margin-right: 5px;
                      cursor: pointer;
                      width: 100px;
                      padding: 10px;
                      text-decoration: none;"
                      onclick="document.getElementById('ToastID').style.display='block'; document.getElementById('ToastID').style.opacity='100';" >
<!-- Toast Conatiner -->
                        <div id="ToastID" class="toast-container" role="alert" style="box-shadow: 0 2px 2px rgba(0, 0, 0, 0.2); border: 1px solid #ddd; position: fixed;bottom: 0; right: 0; width: fit-content; max-width: 25%; margin: 10px; border-radius: 5px; background-color: white;">
<!-- Toast Header -->
                          <div class="toast-header" style="border-bottom: 0.5px solid lightgray; border-radius: 5px; position: relative;padding: 10px; margin-top: -1%;">
<!-- Toast Logo -->

<!-- Change the location of the file as found in TP Book -->

                            <img src="https://tp-book-prod.azurewebsites.net/Filemanager/idcampaing/image-example2.png" class="Toast-logo" alt="..." width="10px" height="10px">

<!-- Change the title of the card inside the strong tag -->

                            <strong class="title-Toast">Toast title</strong>
<!-- Toast Subtitle -->

                            <small class="Subtitle-toast" style="color:rgb(202, 201, 201);">11 mins ago</small>
                            <input value="×"  type="button" class="bttn-close" style="right: 2%; top: 20%; position: absolute; border: none; background:none; font-size: 18px; color: rgb(202, 201, 201);" onclick="document.getElementById('ToastID').style.display='none'; document.getElementById('ToastID').style.opacity='0';" />
                          </div>
